一个拥有各种功能的选择器。

引入

import {SelectModule} from '~/cherry-ui;

基础用法

选择单个选项的选择器
<cherry-select style="width: 120px" [(ngModel)]="selectedOption" [uiPlaceHolder]="'choose option'" [uiAllowClear]="true"> <cherry-option *ngFor="let option of options" [uiLabel]="option.label" [uiValue]="option" [uiDisabled]="option.disabled"> </cherry-option> </cherry-select> <cherry-select style="width: 120px;" [(ngModel)]="selectedOption" [uiPlaceHolder]="'choose option'" [uiDisabled]="true"> <cherry-option *ngFor="let option of options" [uiLabel]="option.label" [uiValue]="option" [uiDisabled]="option.disabled"> </cherry-option> </cherry-select> options = [ { value: 'jack', label: 'Jack' }, { value: 'lucy', label: 'Lucy' }, { value: 'Mike', label: 'Mike' }, { value: 'Lily', label: 'Lily' }, { value: 'disabled', label: 'Disabled', disabled: true }]; selectedOption = this.options[0];

尺寸大小

使用uiSize来选择不同尺寸的选择器
<cherry-select style="width: 120px;" [(ngModel)]="selectedOption" [uiPlaceHolder]="'choose option'" [uiSize]="'sm'"> <cherry-option *ngFor="let option of options" [uiLabel]="option.label" [uiValue]="option" [uiDisabled]="option.disabled"> <cherry-option> </cherry-select> <cherry-select style="width: 120px;" [(ngModel)]="selectedOption" [uiPlaceHolder]="'choose option'"> <cherry-option *ngFor="let option of options" [uiLabel]="option.label" [uiValue]="option" [uiDisabled]="option.disabled"> </cherry-option> </cherry-select> <cherry-select style="width: 120px;" [(ngModel)]="selectedOption" [uiPlaceHolder]="'choose option'" [uiSize]="'lg'"> <cherry-option *ngFor="let option of options" [uiLabel]="option.label" [uiValue]="option" [uiDisabled]="option.disabled"> </cherry-option> </cherry-select>

多选

设置uiMode为multiple来使用多选的选择器
<cherry-select style="width: 240px;" [(ngModel)]="selectedOption" [uiMode]="'multiple'" [uiPlaceHolder]="'choose option'" [uiAllowClear]="true" [uiSize]="'lg'"> <cherry-option *ngFor="let option of options" [uiLabel]="option.label" [uiValue]="option" [uiDisabled]="option.disabled"> </cherry-option> </cherry-select>

新增选项

可以新增选项的选择器 将uiTags启用
<cherry-select style="width: 240px;" [(ngModel)]="selectedOption" [uiMode]="'tags'" [uiPlaceHolder]="'请输入选项并回车'" [uiSize]="'lg'"> <cherry-option *ngFor="let option of options" [uiLabel]="option.label" [uiValue]="option" [uiDisabled]="option.disabled"> </cherry-option> </cherry-select>

自定义option模版

传入自定义模版时 默认模版将不显示

自定义的内容
<cherry-select style="width: 120px;" [(ngModel)]="selectedOption" [uiPlaceHolder]="'自定义模版'"> <cherry-option *ngFor="let option of options" [uiLabel]="option.label" [uiValue]="option"> <ng-template #uiOptionTemplate>自定义的内容</ng-template> </cherry-option> </cherry-select>

滚动到底部的事件

可以定义滚动到顶部的事件,如加载等

Loading...
<cherry-select style="width: 120px;" [(ngModel)]="selectedOption" [uiPlaceHolder]="'滚到底部加载选项'" (uiScrollToBottom)="scrollToBottom()"> <cherry-option *ngFor="let option of options" [uiLabel]="option.label" [uiValue]="option"> </cherry-option> <cherry-option [uiLabel]="'disabled'" [uiDisabled]="true" [uiValue]="'disabled'" *ngIf="loading"> <ng-template #uiOptionTemplate> <i class="fa fa-spinner fa-pulse"></i> Loading... </ng-template> </cherry-option> </cherry-select> scrollToBottom() { if (!this.loading) { this.loading = true; setTimeout(() => { this.loading = false; }, 3000); } }

搜索

输入关键字可以对选项进行搜索并显示搜索结果
<cherry-select style="width: 120px;" [(ngModel)]="selectedOption" [uiPlaceHolder]="'输入关键字'" [uiShowSearch]="true"> <cherry-option *ngFor="let option of options" [uiLabel]="option.label" [uiValue]="option"> </cherry-option> </cherry-select>

API

cherry-select

参数 说明 类型 默认值
ngModel 指定当前选中的条目，支持双向绑定 Array（多选）/String/Object（单选 -
uiPlaceHolder 默认文字 String -
uiAllowClear 可以清除选择框中的已选项 单选下有效 Boolean false
uiDisabled 是否禁用选择框 Boolean false
uiSize 选择框尺寸 'sm'|'lg' -
uiMode 选择多选或者新增多选 'multiple' | 'tags' -
uiScrollToBottom 下拉菜单滚动到底部的事件回调 Func -
uiOpenChange 下拉菜单打开或关闭的事件回调 Func -
uiShowSearch 是否启用搜索功能 Boolean false
uiFilter 是否根据输入过滤选项 Boolean true

cherry-option

参数 说明 类型 默认值
uiLabel 选项文本 String -
uiValue 选项value String/Object -
uiDisabled 是否禁用选项 Boolean false
#uiOptionTemplate 自定义选项模版 ng-template -
