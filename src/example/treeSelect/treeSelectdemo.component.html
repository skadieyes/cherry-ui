可以对树进行选择的选择器

引入

import {TreeSelectModule} from '~/cherry-ui;

基础用法

单选的树选择器
<cherry-tree-select style="width: 250px;" [(ngModel)]="selectedOption" [value]="options" [uiPlaceHolder]="'choose option'"> </cherry-tree-select> options: any; selectedOption: any; ngOnInit() { this.options = [ { label: 'Documents', data: 'Documents Folder', children: [{ label: 'Work', data: 'Work Folder', children: [{ label: 'Expenses', data: 'Expenses Document' }, { label: 'Resume', data: 'Resume Document' }] } ... ] } ]; selectedOption = this.options[1];

多选

支持进行多选的树选择器
<cherry-tree-select style="width: 250px;" [(ngModel)]="selectedOption" [value]="options" [uiMode]="'multiple'" [uiPlaceHolder]="'choose option'"> </cherry-tree-select> options: any; selectedOption: any; ngOnInit() { this.options = [ { label: 'Documents', data: 'Documents Folder', children: [{ label: 'Work', data: 'Work Folder', children: [{ label: 'Expenses', data: 'Expenses Document' }, { label: 'Resume', data: 'Resume Document' }] } ... ] } ]; selectedOption = [this.options[0], this.options[0]];;

限制最大tag数

通过uiMaxTags来限制选择框最多显示多少个已选择项
<cherry-tree-select style="width: 250px;" [(ngModel)]="selectedOption" [value]="options" [uiMaxTags]="2" [uiMode]="'multiple'" [uiPlaceHolder]="'choose option'"> </cherry-tree-select>

显示图标

通过expandedIcon和collapsedIcon来设置每个节点的收起展开图标
<cherry-tree-select style="width: 250px;" [(ngModel)]="selectedOption" [value]="options" [uiPlaceHolder]="'choose option'"> </cherry-tree-select> options: any; selectedOption: any; ngOnInit() { this.options = [ { label: 'Documents', data: 'Documents Folder', expandedIcon: 'fa-folder-open-o', collapsedIcon: 'fa-folder-o', expanded: true, children: [{ label: 'Work', data: 'Work Folder', expandedIcon: 'fa-folder-open-o', collapsedIcon: 'fa-folder-o', expanded: true, children: [{ label: 'Expenses', icon: 'fa-file-o', data: 'Expenses Document' }, { label: 'Resume', icon: 'fa-file-o', data: 'Resume Document' }] } ... ] } ]; selectedOption = this.options[1];

API

cherry-tree-select

参数 说明 类型 默认值
value 传入树选择组件的数据 Array -
ngModel 指定当前选中的条目，支持双向绑定 any -
uiPlaceHolder 默认文字 String -
uiMode 设置树的选择方式为多选或单选 'single'|'multiple' -
uiMaxTags 最大tag数,仅多选下有效 Number -
uiOpenChange 选择框打开或关闭的事件回调 Func -
